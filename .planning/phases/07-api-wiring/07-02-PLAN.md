    ```typescript
    declare module 'next-auth' {
      interface User {
        walletAddress: string;
        username: string;
        profilePictureUrl: string;
        needsOnboarding: boolean;
      }

      interface Session {
        user: {

      }
    }

    declare module 'next-auth/jwt' {

        userId?: string;
        walletAddress?: string;
        username?: string;
        profilePictureUrl?: string;
        needsOnboarding?: boolean;
      }
    }
    ```

    **2. Authorize callback — add provisioning call after SIWE verification succeeds (per D-01, D-04, D-07, D-16):**

    After the line `const userInfo = await MiniKit.getUserInfo(finalPayload.address);`, add:

    ```typescript
    const walletAddress = result.siweMessageData.address;

    // D-01: Provision user in backend
    const apiUrl = process.env.NEXT_PUBLIC_API_URL ?? 'http://localhost:3001';
    const provisionRes = await fetch(`${apiUrl}/api/users/provision`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        walletAddress,
        displayName: userInfo?.username ?? null,
      }),
    });

    // D-04: If provisioning fails, fail auth entirely
    if (!provisionRes.ok) {
      console.error('[auth] provisioning failed:', await provisionRes.text());
      return null;
    }

    const { userId, needsOnboarding } = await provisionRes.json();
    ```

    Then change the return to:
    ```typescript
    return {
      id: userId,                              // D-13: UUID from DB
      walletAddress,                           // D-13: separate field
      needsOnboarding,                         // D-15: computed flag
      username: userInfo?.username ?? '',
      profilePictureUrl: userInfo?.profilePictureUrl ?? '',
    };
    ```

    **3. JWT callback — store all fields including needsOnboarding:**

    ```typescript
    async jwt({ token, user }) {
      if (user) {
        token.userId = user.id;                     // UUID
        token.walletAddress = user.walletAddress;   // 0x address
        token.username = user.username;
        token.profilePictureUrl = user.profilePictureUrl;
        token.needsOnboarding = user.needsOnboarding;
      }
      return token;
    },
    ```

    **4. Session callback — expose all fields:**

    ```typescript
    session: async ({ session, token }) => {
      if (token.userId) {
        session.user.id = token.userId as string;
        session.user.walletAddress = token.walletAddress as string;
        session.user.username = (token.username as string) ?? '';
        session.user.profilePictureUrl = (token.profilePictureUrl as string) ?? '';
        session.user.needsOnboarding = (token.needsOnboarding as boolean) ?? false;
      }
      return session;
    },
    ```

    **Important:** Do NOT move the provisioning call to the jwt callback. The `user` object is only present during initial sign-in (Pitfall 1 from research). The jwt callback runs on every request — provisioning there would fire on every page load.
